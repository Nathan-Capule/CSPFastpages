listTwo = [92,43,74,66,30,12,1]
listThree = [7,13,96,111,33,84,60]
numLists = [listOne, listTwo, listThree]
for x in range(len(numLists)):
    numLists[x].sort()
    middle = int(len(numLists[x])/2)
